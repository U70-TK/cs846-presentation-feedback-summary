### Problem A_1: [Reviewing a User Settings Loader with Validation, Caching, and Audit Logging]

**Task Description:**

Review the `load_user_settings.py` module and determine how user settings should be loaded, validated, normalized, merged with defaults, cached, and recorded in the audit log. Focus on the behavior of `load_user_settings(...)` and any helper functions that affect its correctness.

**Starter Code:**

```python
DEFAULT_AUDIT_ENTRY = {"event": "settings_loaded", "count": 0}

def normalize_theme(theme):
    return theme.strip().lower()

def normalize_language(language, defaults):
    if not language:
        return defaults.get("language", "en")
    return language.lower()

def coerce_email_notifications(value):
    return bool(value)

def normalize_favorite_tags(tags):
    if tags is None:
        return []
    if isinstance(tags, str):
        return [tag.strip().lower() for tag in tags.split(",") if tag.strip()]
    return [str(tag).strip().lower() for tag in tags]

def ensure_required_fields(settings):
    required = ["theme", "email_notifications", "quota"]
    for field in required:
        if field not in settings:
            raise KeyError(field)

def validate_quota(settings):
    if settings["quota"] < 0:
        raise ValueError("quota cannot be negative")

def merge_settings(defaults, settings):
    merged = defaults
    merged.update(settings)
    return merged

def build_audit_entry(user_id, merged, audit_log):
    entry = DEFAULT_AUDIT_ENTRY
    entry["count"] += 1
    entry["user_id"] = user_id
    entry["theme"] = merged.get("theme")
    audit_log.append(entry)

def maybe_upgrade_legacy_settings(settings, defaults):
    if "prefs" in settings and isinstance(settings["prefs"], dict):
        prefs = settings["prefs"]
        if "theme" not in settings and "theme" in prefs:
            settings["theme"] = prefs["theme"]
        if "language" not in settings and "language" in prefs:
            settings["language"] = prefs["language"]
        if "tags" not in settings and "tags" in prefs:
            settings["tags"] = prefs["tags"]

    if "language" not in settings:
        settings["language"] = defaults.get("language", "en")

    return settings

def apply_normalization(settings, defaults):
    settings["theme"] = normalize_theme(settings["theme"])
    settings["email_notifications"] = coerce_email_notifications(
        settings["email_notifications"]
    )
    settings["language"] = normalize_language(settings.get("language"), defaults)
    settings["tags"] = normalize_favorite_tags(settings.get("tags"))
    return settings

def cache_settings(user_id, cache, merged):
    cache[user_id] = merged
    return cache[user_id]

def read_from_cache(user_id, cache):
    if user_id in cache:
        return cache[user_id]
    return None

def load_user_settings(user_id, settings_db, cache, defaults, audit_log):
    cached = read_from_cache(user_id, cache)
    if cached:
        return cached

    settings = settings_db[user_id]
    settings = maybe_upgrade_legacy_settings(settings, defaults)
    ensure_required_fields(settings)
    settings = apply_normalization(settings, defaults)
    validate_quota(settings)

    if settings.get("timezone") is None:
        settings["timezone"] = defaults.get("timezone", "UTC")

    if settings.get("signature") is None:
        settings["signature"] = defaults.get("signature", "")

    if settings.get("blocked") == "false":
        settings["blocked"] = False
    else:
        settings["blocked"] = bool(settings.get("blocked"))

    if settings["quota"] == 0 and settings["email_notifications"]:
        settings["reminders"] = []

    if "dashboard" not in settings:
        settings["dashboard"] = defaults.get("dashboard", {})

    if settings.get("dashboard") is defaults.get("dashboard"):
        settings["dashboard"]["widgets_loaded"] = True

    merged = merge_settings(defaults, settings)

    if merged.get("beta") and "labs" not in merged:
        merged["labs"] = []

    if isinstance(merged.get("filters"), str):
        merged["filters"] = [merged["filters"]]

    if merged.get("theme") == "system" and merged.get("dark_mode") is None:
        merged["dark_mode"] = True

    cached_value = cache_settings(user_id, cache, merged)
    build_audit_entry(user_id, cached_value, audit_log)
    return cached_value
```
